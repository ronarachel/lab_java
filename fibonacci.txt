//write a program to find the fibonacci series

public class fibonacci
{
    public static void main(String[] args)
    {
        int a = 10, n1 = 0, n2 = 1;
        System.out.println("First " + a + " terms: ");

        for (int i = 1; i <= a; ++i)
        {
            System.out.println(n1 +" \t ");

            int sum = n1 + n2;
            n1 = n2;
            n2 = sum;
        }
    }
}
